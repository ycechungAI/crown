50 inline Color4 from_rgba(uint32_t rgba)
51 {
52  Color4 c;
53  c.x = 1.0f/255.0f * ((rgba & 0xff000000) >> 24);
54  c.y = 1.0f/255.0f * ((rgba & 0x00ff0000) >> 16);
55  c.z = 1.0f/255.0f * ((rgba & 0x0000ff00) >> 8);
56  c.w = 1.0f/255.0f * ((rgba & 0x000000ff) >> 0);
57  return c;
58 }
