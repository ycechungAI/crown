124  template <typename T>
125  inline void set_capacity(Vector<T>& v, u32 capacity)
126  {
127  if (capacity == v._capacity)
128  return;
129 
130  if (capacity < v._size)
131  resize(v, capacity);
132 
133  if (capacity > 0)
134  {
135  T* tmp = v._data;
136  v._capacity = capacity;
137  v._data = (T*)v._allocator->allocate(capacity * sizeof(T), alignof(T));
138 
139  for (u32 i = 0; i < v._size; ++i)
140  new (v._data + i) T(tmp[i]);
141 
142  for (u32 i = 0; i < v._size; ++i)
143  tmp[i].~T();
144 
145  v._allocator->deallocate(tmp);
146  }
147  }
