150  bool push_event(const OsEvent& ev)
151  {
152  int cur_tail = _tail.load();
153  int next_tail = increment(cur_tail);
154 
155  if (next_tail != _head.load())
156  {
157  _queue[cur_tail] = ev;
158  _tail.store(next_tail);
159  return true;
160  }
161 
162  return false;
163  }
164 
165  bool pop_event(OsEvent& ev)
166  {
167  const int cur_head = _head.load();
168 
169  if (cur_head == _tail.load())
170  return false;
171 
172  ev = _queue[cur_head];
173  _head.store(increment(cur_head));
174 
175  return true;
176  }
177 
178  int increment(int idx) const
179  {
180  return (idx + 1) % MAX_OS_EVENTS;
181  }
