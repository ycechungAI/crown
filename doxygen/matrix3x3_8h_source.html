25 inline Matrix3x3 matrix3x3(const Quaternion& r)
26 {
27  Matrix3x3 m;
28  m.x.x = 1.0f - 2.0f * r.y * r.y - 2.0f * r.z * r.z;
29  m.x.y = 2.0f * r.x * r.y + 2.0f * r.w * r.z;
30  m.x.z = 2.0f * r.x * r.z - 2.0f * r.w * r.y;
31 
32  m.y.x = 2.0f * r.x * r.y - 2.0f * r.w * r.z;
33  m.y.y = 1.0f - 2.0f * r.x * r.x - 2.0f * r.z * r.z;
34  m.y.z = 2.0f * r.y * r.z + 2.0f * r.w * r.x;
35 
36  m.z.x = 2.0f * r.x * r.z + 2.0f * r.w * r.y;
37  m.z.y = 2.0f * r.y * r.z - 2.0f * r.w * r.x;
38  m.z.z = 1.0f - 2.0f * r.x * r.x - 2.0f * r.y * r.y;
39  return m;
40 }

182 inline Matrix3x3& invert(Matrix3x3& m)
183 {
184  Matrix3x3 mat;
185 
186  mat.x.x = (m.y.y * m.z.z - m.z.y * m.y.z);
187  mat.x.y = (m.x.y * m.z.z - m.z.y * m.x.z);
188  mat.x.z = (m.x.y * m.y.z - m.y.y * m.x.z);
189 
190  const float inv_det = 1.0f / (m.x.x * mat.x.x - m.y.x * mat.x.y + m.z.x * mat.x.z);
191 
192  mat.y.x = (m.y.x * m.z.z - m.z.x * m.y.z);
193  mat.y.y = (m.x.x * m.z.z - m.z.x * m.x.z);
194  mat.y.z = (m.x.x * m.y.z - m.y.x * m.x.z);
195  mat.z.x = (m.y.x * m.z.y - m.z.x * m.y.y);
196  mat.z.y = (m.x.x * m.z.y - m.z.x * m.x.y);
197  mat.z.z = (m.x.x * m.y.y - m.y.x * m.x.y);
198 
199  m.x.x = + mat.x.x * inv_det;
200  m.x.y = - mat.x.y * inv_det;
201  m.x.z = + mat.x.z * inv_det;
202  m.y.x = - mat.y.x * inv_det;
203  m.y.y = + mat.y.y * inv_det;
204  m.y.z = - mat.y.z * inv_det;
205  m.z.x = + mat.z.x * inv_det;
206  m.z.y = - mat.z.y * inv_det;
207  m.z.z = + mat.z.z * inv_det;
208 
209  return m;
210 }
