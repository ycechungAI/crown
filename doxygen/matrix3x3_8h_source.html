28 inline Matrix3x3 matrix3x3(const Quaternion& r)
29 {
30  Matrix3x3 m;
31  m.x.x = 1.0f - 2.0f * r.y * r.y - 2.0f * r.z * r.z;
32  m.x.y = 2.0f * r.x * r.y + 2.0f * r.w * r.z;
33  m.x.z = 2.0f * r.x * r.z - 2.0f * r.w * r.y;
34 
35  m.y.x = 2.0f * r.x * r.y - 2.0f * r.w * r.z;
36  m.y.y = 1.0f - 2.0f * r.x * r.x - 2.0f * r.z * r.z;
37  m.y.z = 2.0f * r.y * r.z + 2.0f * r.w * r.x;
38 
39  m.z.x = 2.0f * r.x * r.z + 2.0f * r.w * r.y;
40  m.z.y = 2.0f * r.y * r.z - 2.0f * r.w * r.x;
41  m.z.z = 1.0f - 2.0f * r.x * r.x - 2.0f * r.y * r.y;
42  return m;
43 }

185 inline Matrix3x3& invert(Matrix3x3& m)
186 {
187  Matrix3x3 mat;
188 
189  mat.x.x = (m.y.y * m.z.z - m.z.y * m.y.z);
190  mat.x.y = (m.x.y * m.z.z - m.z.y * m.x.z);
191  mat.x.z = (m.x.y * m.y.z - m.y.y * m.x.z);
192 
193  const float inv_det = 1.0f / (m.x.x * mat.x.x - m.y.x * mat.x.y + m.z.x * mat.x.z);
194 
195  mat.y.x = (m.y.x * m.z.z - m.z.x * m.y.z);
196  mat.y.y = (m.x.x * m.z.z - m.z.x * m.x.z);
197  mat.y.z = (m.x.x * m.y.z - m.y.x * m.x.z);
198  mat.z.x = (m.y.x * m.z.y - m.z.x * m.y.y);
199  mat.z.y = (m.x.x * m.z.y - m.z.x * m.x.y);
200  mat.z.z = (m.x.x * m.y.y - m.y.x * m.x.y);
201 
202  m.x.x = + mat.x.x * inv_det;
203  m.x.y = - mat.x.y * inv_det;
204  m.x.z = + mat.x.z * inv_det;
205  m.y.x = - mat.y.x * inv_det;
206  m.y.y = + mat.y.y * inv_det;
207  m.y.z = - mat.y.z * inv_det;
208  m.z.x = + mat.z.x * inv_det;
209  m.z.y = - mat.z.y * inv_det;

239 inline Quaternion rotation(const Matrix3x3& m)
240 {
241  const float ww = m.x.x + m.y.y + m.z.z;
242  const float xx = m.x.x - m.y.y - m.z.z;
243  const float yy = m.y.y - m.x.x - m.z.z;
244  const float zz = m.z.z - m.x.x - m.y.y;
245  float max = ww;
246  uint32_t index = 0;
247 
248  if (xx > max)
249  {
250  max = xx;
251  index = 1;
252  }
253 
254  if (yy > max)
255  {
256  max = yy;
257  index = 2;
258  }
259 
260  if (zz > max)
261  {
262  max = zz;
263  index = 3;
264  }
265 
266  const float biggest = sqrt(max + 1.0f) * 0.5f;
267  const float mult = 0.25f / biggest;
268 
269  Quaternion tmp;
270  switch (index)
271  {
272  case 0:
273  {
274  tmp.w = biggest;
275  tmp.x = (m.y.z - m.z.y) * mult;
276  tmp.y = (m.z.x - m.x.z) * mult;
277  tmp.z = (m.x.y - m.y.x) * mult;
278  break;
279  }
280  case 1:
281  {
282  tmp.x = biggest;
283  tmp.w = (m.y.z - m.z.y) * mult;
284  tmp.y = (m.x.y + m.y.x) * mult;
285  tmp.z = (m.z.x + m.x.z) * mult;
286  break;
287  }
288  case 2:
289  {
290  tmp.y = biggest;
291  tmp.w = (m.z.x - m.x.z) * mult;
292  tmp.x = (m.x.y + m.y.x) * mult;
293  tmp.z = (m.y.z + m.z.y) * mult;
294  break;
295  }
296  case 3:
297  {
298  tmp.z = biggest;
299  tmp.w = (m.x.y - m.y.x) * mult;
300  tmp.x = (m.z.x + m.x.z) * mult;
301  tmp.y = (m.y.z + m.z.y) * mult;
302  break;
303  }
304  default:
305  {
306  CE_FATAL("You should not be here");
307  break;
308  }
309  }
310 
311  normalize(tmp);
312  return tmp;
313 }
