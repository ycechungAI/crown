131 inline Quaternion power(const Quaternion& q, f32 exp)
132 {
133  if (fabs(q.w) < 0.9999)
134  {
135  const f32 alpha = acos(q.w); // alpha = theta/2
136  const f32 new_alpha = alpha * exp;
137  const f32 mult = sinf(new_alpha) / sinf(alpha);
138 
139  Quaternion r;

141  r.x = q.x * mult;

143  r.z = q.z * mult;

185 inline Quaternion lerp(const Quaternion& a, const Quaternion& b, f32 t)
186 {
187  const f32 t1 = 1.0f - t;
188 
189  Quaternion r;
190 
191  if (dot(a, b) < 0.0f)
192  {
193  r.x = t1*a.x + t*-b.x;
194  r.y = t1*a.y + t*-b.y;
195  r.z = t1*a.z + t*-b.z;
196  r.w = t1*a.w + t*-b.w;
197  }
198  else
199  {
200  r.x = t1*a.x + t*b.x;
201  r.y = t1*a.y + t*b.y;
202  r.z = t1*a.z + t*b.z;
203  r.w = t1*a.w + t*b.w;
204  }
205 
206  return normalize(r);
207 }
