67  template <typename TKey, typename TValue, typename Hash>
68  inline u32 probe_distance(const HashMap<TKey, TValue, Hash>& m, u32 hash, u32 slot_index)
69  {
70  const u32 hash_i = hash & m._mask;
71  return (slot_index + m._capacity - hash_i) & m._mask;

74  template <typename TKey, typename TValue, typename Hash>
75  u32 find(const HashMap<TKey, TValue, Hash>& m, const TKey& key)
76  {
77  if (m._size == 0)
78  return END_OF_LIST;
79 
80  const u32 hash = hash_key<TKey, Hash>(key);
81  u32 hash_i = hash & m._mask;
82  u32 dist = 0;
83  for(;;)
84  {
85  if (m._index[hash_i].index == FREE)
86  return END_OF_LIST;
87  else if (dist > probe_distance(m, m._index[hash_i].hash, hash_i))
88  return END_OF_LIST;
89  else if (!is_deleted(m._index[hash_i].index) && m._index[hash_i].hash == hash && m._data[hash_i].pair.first == key)
90  return hash_i;
91 
92  hash_i = (hash_i + 1) & m._mask;
93  ++dist;
94  }
95  }

97  template <typename TKey, typename TValue, typename Hash>
98  void insert(HashMap<TKey, TValue, Hash>& m, u32 hash, const TKey& key, const TValue& value)
99  {
100  PAIR(TKey, TValue) new_item(*m._allocator);
101  new_item.first = key;
102  new_item.second = value;
103 
104  u32 hash_i = hash & m._mask;
105  u32 dist = 0;
106  for(;;)
107  {
108  if (m._index[hash_i].index == FREE)
109  goto INSERT_AND_RETURN;
110 
111  // If the existing elem has probed less than us, then swap places with existing
112  // elem, and keep going to find another slot for that elem.
113  u32 existing_elem_probe_dist = probe_distance(m, m._index[hash_i].hash, hash_i);
114  if (is_deleted(m._index[hash_i].index) || existing_elem_probe_dist < dist)
115  {
116  if (is_deleted(m._index[hash_i].index))
117  goto INSERT_AND_RETURN;
118 
119  std::swap(hash, m._index[hash_i].hash);
120  m._index[hash_i].index = 0x0123abcd;
121  PAIR(TKey, TValue) tmp(*m._allocator);
122  memcpy(&tmp, &m._data[hash_i].pair, sizeof(new_item));
123  memcpy(&m._data[hash_i].pair, &new_item, sizeof(new_item));
124  memcpy(&new_item, &tmp, sizeof(new_item));
125 
126  dist = existing_elem_probe_dist;
127  }
128 
129  hash_i = (hash_i + 1) & m._mask;
130  ++dist;
131  }
132 
133  INSERT_AND_RETURN:
134  new (m._data + hash_i) typename HashMap<TKey, TValue, Hash>::Entry(*m._allocator);
135  memcpy(m._data + hash_i, &new_item, sizeof(new_item));
136  m._index[hash_i].hash = hash;
137  m._index[hash_i].index = 0x0123abcd;
138  PAIR(TKey, TValue) empty(*m._allocator);
139  memcpy(&new_item, &empty, sizeof(new_item));
140  }

142  template <typename TKey, typename TValue, typename Hash>
143  void rehash(HashMap<TKey, TValue, Hash>& m, u32 new_capacity)
144  {
145  typedef typename HashMap<TKey, TValue, Hash>::Entry Entry;
146  typedef typename HashMap<TKey, TValue, Hash>::Index Index;
147 
148  HashMap<TKey, TValue, Hash> nm(*m._allocator);
149  nm._index = (Index*)nm._allocator->allocate(new_capacity*sizeof(Index), alignof(Index));
150  nm._data = (Entry*)nm._allocator->allocate(new_capacity*sizeof(Entry), alignof(Entry));
151 
152  // Flag all elements as free
153  for (u32 i = 0; i < new_capacity; ++i)
154  {
155  nm._index[i].hash = 0;
156  nm._index[i].index = FREE;
157  }
158 
159  nm._capacity = new_capacity;
160  nm._size = m._size;
161  nm._mask = new_capacity - 1;
162 
163  for (u32 i = 0; i < m._capacity; ++i)
164  {
165  typename HashMap<TKey, TValue, Hash>::Entry& e = m._data[i];
166  const u32 hash = m._index[i].hash;
167  const u32 index = m._index[i].index;
168 
169  if (index != FREE && !is_deleted(index))
170  hash_map_internal::insert(nm, hash, e.pair.first, e.pair.second);
171  }
172 
173  HashMap<TKey, TValue, Hash> empty(*m._allocator);
174  m.~HashMap<TKey, TValue, Hash>();
175  memcpy(&m, &nm, sizeof(HashMap<TKey, TValue, Hash>));
176  memcpy(&nm, &empty, sizeof(HashMap<TKey, TValue, Hash>));
177  }

223  template <typename TKey, typename TValue, typename Hash>
224  void set(HashMap<TKey, TValue, Hash>& m, const TKey& key, const TValue& value)
225  {
226  if (m._capacity == 0)
227  hash_map_internal::grow(m);
228 
229  // Find or make
230  const u32 i = hash_map_internal::find(m, key);
231  if (i == hash_map_internal::END_OF_LIST)
232  {
233  hash_map_internal::insert(m, hash_map_internal::hash_key<TKey, Hash>(key), key, value);
234  ++m._size;
235  }
236  else
237  {
238  m._data[i].pair.second = value;
239  }
240  if (hash_map_internal::full(m))
241  hash_map_internal::grow(m);
242  }

245  void remove(HashMap<TKey, TValue, Hash>& m, const TKey& key)
246  {
247  const u32 i = hash_map_internal::find(m, key);
248  if (i == hash_map_internal::END_OF_LIST)
249  return;
