77  template <typename TKey, typename TValue, typename Compare>
78  inline FindResult find(const SortMap<TKey, TValue, Compare>& m, const TKey& key)
79  {
80  CE_ASSERT(m._is_sorted, "Map not sorted");
81 
82  FindResult result;
83  result.item_i = END_OF_LIST;
84 
85  const typename SortMap<TKey, TValue, Compare>::Entry* first =
86  std::lower_bound(vector::begin(m._data), vector::end(m._data), key,
87  sort_map_internal::CompareEntry<TKey, TValue, Compare>());
88 
89  if (first != vector::end(m._data) && !(key < first->first))
90  result.item_i = u32(first - vector::begin(m._data));
91 
92  return result;
93  }

161  {

163 
164  if (result.item_i == sort_map_internal::END_OF_LIST)
165  return;

167  if (vector::size(m._data))
