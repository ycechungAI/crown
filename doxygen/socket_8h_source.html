219  AcceptResult accept_internal(TCPSocket& c)
220  {

222 
223  AcceptResult ar;
224  ar.error = AcceptResult::NO_ERROR;
225 
226 #if CROWN_PLATFORM_POSIX
227  if (err >= 0)
228  c._socket = err;
229  else if (err == -1 && errno == EBADF)
230  ar.error = AcceptResult::BAD_SOCKET;
231  else if (err == -1 && (errno == EAGAIN || errno == EWOULDBLOCK))
232  ar.error = AcceptResult::NO_CONNECTION;
233  else
234  ar.error = AcceptResult::UNKNOWN;
235 #elif CROWN_PLATFORM_WINDOWS
236  if (err != INVALID_SOCKET)
237  {
238  c._socket = err;
239  return ar;
240  }
241 
242  int wsaerr = WSAGetLastError();
243  if (wsaerr == WSAEWOULDBLOCK)
244  ar.error = AcceptResult::NO_CONNECTION;
245  else
246  ar.error = AcceptResult::UNKNOWN;
247 #endif

265  ReadResult read_internal(void* data, u32 size)
266  {
267  ReadResult rr;
268  rr.error = ReadResult::NO_ERROR;
269  rr.bytes_read = 0;
270 
271  char* buf = (char*)data;
272  u32 to_read = size;
273 
274  while (to_read > 0)
275  {
276 #if CROWN_PLATFORM_POSIX
277  ssize_t read_bytes = ::recv(_socket, buf, to_read, 0);
278 
279  if (read_bytes == -1 && (errno == EAGAIN || errno == EWOULDBLOCK))
280  {
281  return rr;
282  }
283  else if (read_bytes == -1 && errno == ETIMEDOUT)
284  {
285  rr.error = ReadResult::TIMEOUT;
286  return rr;
287  }
288  else if (read_bytes == 0)
289  {
290  rr.error = ReadResult::REMOTE_CLOSED;
291  return rr;
292  }
293 #elif CROWN_PLATFORM_WINDOWS
294  int read_bytes = ::recv(_socket, buf, (int)to_read, 0);
295 
296  if (read_bytes == SOCKET_ERROR && WSAGetLastError() == WSAEWOULDBLOCK)
297  return rr;
298  else if (read_bytes == SOCKET_ERROR && WSAGetLastError() == WSAETIMEDOUT)
299  {
300  rr.error = ReadResult::TIMEOUT;
301  return rr;
302  }
303  else if (read_bytes == 0)
304  {
305  rr.error = ReadResult::REMOTE_CLOSED;
306  return rr;
307  }
308 #endif
309  buf += read_bytes;
310  to_read -= read_bytes;
311  rr.bytes_read += read_bytes;
312  }
313 
314  return rr;
315  }
