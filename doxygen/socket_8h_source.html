233  ReadResult read_internal(void* data, u32 size)
234  {
235  ReadResult rr;
236  rr.error = ReadResult::SUCCESS;
237  rr.bytes_read = 0;
238 
239  u32 to_read = size;
240 
241  while (to_read > 0)
242  {
243  int bytes_read = ::recv(_socket
244  , (char*)data + rr.bytes_read
245  , to_read
246  , 0
247  );
248 
249  if (bytes_read == SOCKET_ERROR)
250  {
251  if (last_error() == EWOULDBLOCK)
252  rr.error = ReadResult::WOULDBLOCK;
253  else if (last_error() == ETIMEDOUT)
254  rr.error = ReadResult::TIMEOUT;
255  else
256  rr.error = ReadResult::UNKNOWN;
257  return rr;
258  }
259  else if (bytes_read == 0)
260  {
261  rr.error = ReadResult::REMOTE_CLOSED;
262  return rr;
263  }
264 
265  to_read -= bytes_read;
266  rr.bytes_read += bytes_read;
