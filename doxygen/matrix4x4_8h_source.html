96 inline Matrix4x4 matrix4x4(const Quaternion& r, const Vector3& p)
97 {
98  Matrix4x4 m;
99  m.x.x = 1.0f - 2.0f * r.y * r.y - 2.0f * r.z * r.z;
100  m.x.y = 2.0f * r.x * r.y + 2.0f * r.w * r.z;
101  m.x.z = 2.0f * r.x * r.z - 2.0f * r.w * r.y;
102  m.x.w = 0.0f;
103 
104  m.y.x = 2.0f * r.x * r.y - 2.0f * r.w * r.z;
105  m.y.y = 1.0f - 2.0f * r.x * r.x - 2.0f * r.z * r.z;
106  m.y.z = 2.0f * r.y * r.z + 2.0f * r.w * r.x;
107  m.y.w = 0.0f;
108 
109  m.z.x = 2.0f * r.x * r.z + 2.0f * r.w * r.y;
110  m.z.y = 2.0f * r.y * r.z - 2.0f * r.w * r.x;
111  m.z.z = 1.0f - 2.0f * r.x * r.x - 2.0f * r.y * r.y;
112  m.z.w = 0.0f;
113 
114  m.t.x = p.x;
115  m.t.y = p.y;
116  m.t.z = p.z;
117  m.t.w = 1.0f;

397 {
398  const float m02m07_m06m03 = m.x.z * m.y.w - m.y.z * m.x.w;
399  const float m02m11_m10m03 = m.x.z * m.z.w - m.z.z * m.x.w;
400  const float m02m15_m14m03 = m.x.z * m.t.w - m.t.z * m.x.w;
401  const float m06m11_m10m07 = m.y.z * m.z.w - m.z.z * m.y.w;
402  const float m06m15_m14m07 = m.y.z * m.t.w - m.t.z * m.y.w;
