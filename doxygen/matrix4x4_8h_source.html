102 inline Matrix4x4 matrix4x4(const Quaternion& r, const Vector3& t)
103 {
104  Matrix4x4 m;
105  m.x.x = 1.0f - 2.0f * r.y * r.y - 2.0f * r.z * r.z;
106  m.x.y = 2.0f * r.x * r.y + 2.0f * r.w * r.z;
107  m.x.z = 2.0f * r.x * r.z - 2.0f * r.w * r.y;
108  m.x.w = 0.0f;
109 
110  m.y.x = 2.0f * r.x * r.y - 2.0f * r.w * r.z;
111  m.y.y = 1.0f - 2.0f * r.x * r.x - 2.0f * r.z * r.z;
112  m.y.z = 2.0f * r.y * r.z + 2.0f * r.w * r.x;
113  m.y.w = 0.0f;
114 
115  m.z.x = 2.0f * r.x * r.z + 2.0f * r.w * r.y;
116  m.z.y = 2.0f * r.y * r.z - 2.0f * r.w * r.x;
117  m.z.z = 1.0f - 2.0f * r.x * r.x - 2.0f * r.y * r.y;
118  m.z.w = 0.0f;
119 
120  m.t.x = t.x;
121  m.t.y = t.y;
122  m.t.z = t.z;
123  m.t.w = 1.0f;
124  return m;

324 inline void orthographic(Matrix4x4& m, f32 left, f32 right, f32 bottom, f32 top, f32 nnear, f32 ffar)
325 {
326  m.x.x = 2.0f / (right - left);

328  m.x.z = 0.0f;
329  m.x.w = 0.0f;
330 
331  m.y.x = 0.0f;
332  m.y.y = 2.0f / (top - bottom);
333  m.y.z = 0.0f;
334  m.y.w = 0.0f;
335 
336  m.z.x = 0.0f;
337  m.z.y = 0.0f;
338  m.z.z = 1.0f / (ffar - nnear);
339  m.z.w = 0.0f;
340 
341  m.t.x = (left + right) / (left - right);
342  m.t.y = (top + bottom) / (bottom - top);
343  m.t.z = nnear / (nnear - ffar);
344  m.t.w = 1.0f;
345 }
